// Simple request
fetch('https://evaloncloud.com/?code=' + 
  encodeURIComponent('return 40 + 2'))
  .then(r => r.text())
  .then(console.log) // Output: 42

// With data context
fetch('https://evaloncloud.com/?' + 
  'input=https://api.example.com/data&' +
  'code=' + encodeURIComponent('return data.length'))
  .then(r => r.text())
  .then(console.log)
